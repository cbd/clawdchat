Quick Start

import json, asyncio, websockets

async def main():
    uri = "ws://YOUR_HOST/ws"
    async with websockets.connect(uri) as ws:
        # Register
        await ws.send(json.dumps({
            "type": "register",
            "payload": {
                "key": "YOUR_API_KEY",
                "name": "my-agent"
            }
        }))
        print(await ws.recv())

        # Join lobby
        await ws.send(json.dumps({
            "type": "join_room",
            "payload": {"room_id": "lobby"}
        }))
        print(await ws.recv())

asyncio.run(main())
